51 
52  BigInteger p1 = 0;
53  BigInteger p2 = 0;
54 
55  // Could be cleaned up with tryParse
56  try
57  {
58  p1 = BigInteger.Parse(split[0]);
59  }
60  catch (Exception)
61  {
62  p1 = 0;
63  }
64 
65  try
66  {
67  p2 = BigInteger.Parse(second_part);
68  }
69  catch (Exception)
70  {
71  p2 = 0;
72  }
73 
74  // Check for partial decimals
75  int s2_length = second_part.Length;
76  double exponent = 0;
77  if(s2_length < num_decimals)
78  exponent = num_decimals - s2_length;
79 
80  double multiplier = 1;
81  if(exponent > 0)
82  multiplier = Math.Pow(10, exponent);
83 
84  // Multiply the second part if neccessary
85  p2 = BigInteger.Multiply(p2, new BigInteger(multiplier));
86 
87  // Multiply the first part to make room for the decimals
88  p1 = BigInteger.Multiply(p1, divisor);
89 
90  // Finally, add both parts
91  amount = BigInteger.Add(p1, p2);
92  }
93  else
94  {
95  try
96  {
97  amount = BigInteger.Parse(number);
98  }
99  catch(Exception)

101  amount = 0;

103  // No decimals detected, multiply the amount
