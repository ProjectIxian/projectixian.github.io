344  public byte[] encryptWithChacha(byte[] input, byte[] key)
345  {
346  // Create a buffer that will contain the encrypted output and an 8 byte nonce
347  byte[] outData = new byte[input.Length + 8];
348 
349  // Generate the 8 byte nonce
350  byte[] nonce = new byte[8];
351 
352  using (RNGCryptoServiceProvider rngCsp = new RNGCryptoServiceProvider())
353  {
354  // Fill the array with a random value.
355  rngCsp.GetBytes(nonce);
356  }
357 
358  // Prevent leading 0 to avoid edge cases
359  if (nonce[0] == 0)
360  nonce[0] = 1;
361 
362  // Generate the Chacha engine
363  var parms = new ParametersWithIV(new KeyParameter(key), nonce);
364  var chacha = new ChaChaEngine(chacha_rounds);
365 
366  try
367  {
368  chacha.Init(true, parms);
369  }
370  catch (Exception e)
371  {
372  Logging.error(string.Format("Error in chacha encryption. {0}", e.ToString()));
373  return null;
374  }
375 
376  // Encrypt the input data while maintaing an 8 byte offset at the start
377  chacha.ProcessBytes(input, 0, input.Length, outData, 8);
378 
379  // Copy the 8 byte nonce to the start of outData buffer
380  Buffer.BlockCopy(nonce, 0, outData, 0, 8);
381 
382  // Return the encrypted data buffer
383  return outData;
384  }
