190  public BigInteger(long value)
191  {
192  data = new uint[maxLength];
193  long tempVal = value;
194 
195  // copy bytes from long to BigInteger without any assumption of
196  // the length of the long datatype
197 
198  dataLength = 0;
199  while (value != 0 && dataLength < maxLength)
200  {
201  data[dataLength] = (uint)(value & 0xFFFFFFFF);
202  value >>= 32;
203  dataLength++;
204  }
205 
206  if (tempVal > 0) // overflow check for +ve value
207  {
208  if (value != 0 || (data[maxLength - 1] & 0x80000000) != 0)
209  throw (new ArithmeticException("Positive overflow in constructor."));
210  }
211  else if (tempVal < 0) // underflow check for -ve value
212  {
213  if (value != -1 || (data[dataLength - 1] & 0x80000000) == 0)
214  throw (new ArithmeticException("Negative underflow in constructor."));
215  }
216 
217  if (dataLength == 0)
218  dataLength = 1;

265  //***********************************************************************
266  // Constructor (Default value provided by a string of digits of the
267  // specified base)
268  //
269  // Example (base 10)
270  // -----------------
271  // To initialize "a" with the default value of 1234 in base 10
272  // BigInteger a = new BigInteger("1234", 10)
273  //
274  // To initialize "a" with the default value of -1234
275  // BigInteger a = new BigInteger("-1234", 10)
276  //
277  // Example (base 16)
278  // -----------------
279  // To initialize "a" with the default value of 0x1D4F in base 16
280  // BigInteger a = new BigInteger("1D4F", 16)
281  //
282  // To initialize "a" with the default value of -0x1D4F
283  // BigInteger a = new BigInteger("-1D4F", 16)
284  //
285  // Note that string values are specified in the <sign><magnitude>
286  // format.
287  //
288  //***********************************************************************
289 
290  public BigInteger(string value, int radix)
291  {
292  BigInteger multiplier = new BigInteger(1);
293  BigInteger result = new BigInteger();
294  value = (value.ToUpper()).Trim();
295  int limit = 0;
296 
297  if (value[0] == '-')
298  limit = 1;
299 
300  for (int i = value.Length - 1; i >= limit; i--)
301  {
302  int posVal = (int)value[i];
303 
304  if (posVal >= '0' && posVal <= '9')
305  posVal -= '0';
306  else if (posVal >= 'A' && posVal <= 'Z')
307  posVal = (posVal - 'A') + 10;
308  else
309  posVal = 9999999; // arbitrary large
310 
311 
312  if (posVal >= radix)
313  throw (new ArithmeticException("Invalid string in constructor."));
314  else
315  {
316  if (value[0] == '-')
317  posVal = -posVal;
318 
319  result = result + (multiplier * posVal);
320 
321  if ((i - 1) >= limit)
322  multiplier = multiplier * radix;
323  }
324  }
325 
326  if (value[0] == '-') // negative values
327  {
328  if ((result.data[maxLength - 1] & 0x80000000) == 0)
329  throw (new ArithmeticException("Negative underflow in constructor."));
330  }
331  else // positive values
332  {
333  if ((result.data[maxLength - 1] & 0x80000000) != 0)
334  throw (new ArithmeticException("Positive overflow in constructor."));
335  }
336 
337  data = new uint[maxLength];
338  for (int i = 0; i < result.dataLength; i++)
339  data[i] = result.data[i];
340 
341  dataLength = result.dataLength;
342  }

345  //***********************************************************************
346  // Constructor (Default value provided by an array of bytes)
347  //
348  // The lowest index of the input byte array (i.e [0]) should contain the
349  // most significant byte of the number, and the highest index should
350  // contain the least significant byte.
351  //
352  // E.g.
353  // To initialize "a" with the default value of 0x1D4F in base 16
354  // byte[] temp = { 0x1D, 0x4F };
355  // BigInteger a = new BigInteger(temp)
356  //
357  // Note that this method of initialization does not allow the
358  // sign to be specified.

362  public BigInteger(byte[] inData)
363  {
364  dataLength = inData.Length >> 2;
365 
366  int leftOver = inData.Length & 0x3;
367  if (leftOver != 0) // length not multiples of 4
368  dataLength++;
369 
370 
371  if (dataLength > maxLength)
372  throw (new ArithmeticException("Byte overflow in constructor."));
373 
374  data = new uint[maxLength];
375 
376  for (int i = inData.Length - 1, j = 0; i >= 3; i -= 4, j++)
377  {
378  data[j] = (uint)((inData[i - 3] << 24) + (inData[i - 2] << 16) +
379  (inData[i - 1] << 8) + inData[i]);
380  }
381 
382  if (leftOver == 1)
383  data[dataLength - 1] = (uint)inData[0];
384  else if (leftOver == 2)
385  data[dataLength - 1] = (uint)((inData[0] << 8) + inData[1]);
386  else if (leftOver == 3)
387  data[dataLength - 1] = (uint)((inData[0] << 16) + (inData[1] << 8) + inData[2]);
388 
389 
390  while (dataLength > 1 && data[dataLength - 1] == 0)
391  dataLength--;
392 
393  //Console.WriteLine("Len = " + dataLength);
394  }

493  public static BigInteger operator +(BigInteger bi1, BigInteger bi2)
494  {
495  BigInteger result = new BigInteger();
496 
497  result.dataLength = (bi1.dataLength > bi2.dataLength) ? bi1.dataLength : bi2.dataLength;
498 
499  long carry = 0;
500  for (int i = 0; i < result.dataLength; i++)
501  {
502  long sum = (long)bi1.data[i] + (long)bi2.data[i] + carry;
503  carry = sum >> 32;
504  result.data[i] = (uint)(sum & 0xFFFFFFFF);
505  }
506 
507  if (carry != 0 && result.dataLength < maxLength)
508  {
509  result.data[result.dataLength] = (uint)(carry);
510  result.dataLength++;
511  }
512 
513  while (result.dataLength > 1 && result.data[result.dataLength - 1] == 0)
514  result.dataLength--;
515 
516 
517  // overflow check
518  int lastPos = maxLength - 1;
519  if ((bi1.data[lastPos] & 0x80000000) == (bi2.data[lastPos] & 0x80000000) &&
520  (result.data[lastPos] & 0x80000000) != (bi1.data[lastPos] & 0x80000000))
521  {
522  throw (new ArithmeticException());
523  }
524 
525  return result;
526  }

1675  {

1679  int t = i;
1680  for (int j = 0; j < n.dataLength && t < kPlusOne; j++, t++)
1681  {

1684  (ulong)r2.data[t] + mcarry;
