302  public byte[] calculateWalletStateChecksum(bool snapshot = false)
303  {
304  lock (stateLock)
305  {
306  if (snapshot == false && cachedChecksum != null)
307  {
308  return cachedChecksum;
309  }
310  else if (snapshot == true && cachedDeltaChecksum != null)
311  {
312  return cachedDeltaChecksum;
313  }
314 
315  // TODO: This could get unwieldy above ~100M wallet addresses. We have to implement sharding by then.
316  SortedSet<byte[]> eligible_addresses = null;
317  eligible_addresses = new SortedSet<byte[]>(walletState.Keys, new ByteArrayComparer());
318 
319  if (snapshot == true)
320  {
321  if (wsDelta != null)
322  {
323  foreach (var entry in wsDelta)
324  {
325  eligible_addresses.Add(entry.Key);
326  }
327  }
328  }
329 
330  byte[] checksum = null;
331  if (cachedBlockVersion <= 2)
332  {
333  checksum = Crypto.sha512quTrunc(Encoding.UTF8.GetBytes("IXIAN-DLT" + version));
334  }else
335  {
336  checksum = Crypto.sha512sqTrunc(Encoding.UTF8.GetBytes("IXIAN-DLT" + version), 0, 0, 64);
337  }
338 
339  // TODO: This is probably not the optimal way to do this. Maybe we could do it by blocks to reduce calls to sha256
340  // Note: addresses are not fixed size
341  foreach (byte[] addr in eligible_addresses)
342  {
343  byte[] wallet_checksum = getWallet(addr, snapshot).calculateChecksum(cachedBlockVersion);
344  if (cachedBlockVersion <= 2)
345  {
346  checksum = Crypto.sha512quTrunc(Encoding.UTF8.GetBytes(Crypto.hashToString(checksum) + Crypto.hashToString(wallet_checksum)));
347  }else
348  {
349  List<byte> tmp_hash = checksum.ToList();
350  tmp_hash.AddRange(wallet_checksum);
351  checksum = Crypto.sha512sqTrunc(tmp_hash.ToArray(), 0, 0, 64);
352  }
353  }
354 
355  if (snapshot == false)
356  {
357  cachedChecksum = checksum;
358  }
359  else

362  }
