49  private void twiddleRandomState()
50  {
51  int blocks = currentRandomState.Length / 64;
52  // each 64-byte region separately
53  for (int i = 0; i < blocks; i++)
54  {
55  Array.Copy(sha.ComputeHash(currentRandomState, i * 64, 64), 0, currentRandomState, i * 64, 64);
56  }
57  currentIteration += 1;

215  public static double calculateBytestreamEntropy(byte[] data)
216  {
217  int[] occurences = Enumerable.Repeat(0, 256).ToArray();
218  foreach(byte b in data)
219  {
220  occurences[b]++;
221  }
222  double[] probs = Enumerable.Repeat(0.0, 256).ToArray();
223  for(int i = 0; i < 256; i++)
224  {
225  probs[i] = (double)occurences[i] / (double)data.Length;
226  }
227  return probs.Where(p => p > 0).Select(p => -1* p * Math.Log(p, 2)).Sum() / 8.0;
228  }
