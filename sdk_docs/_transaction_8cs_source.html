528  public bool verifySignature(byte[] pubkey, byte[] nonce)
529  {
530  if(pubkey == null)
531  {
532  return false;
533  }
534 
535  // Skip signature verification for staking rewards
536  if (type == (int)Type.StakingReward || type == (int)Type.Genesis)
537  {
538  return true;
539  }
540 
541  Address p_address = new Address(pubkey, nonce);
542  bool allowed = false;
543  Wallet from_wallet = Node.walletState.getWallet((new Address(this.pubKey)).address);
544  if(from_wallet != null && from_wallet.id.SequenceEqual(p_address.address))
545  {
546  allowed = true;
547  } else if (type == (int)Transaction.Type.MultisigTX || type == (int)Transaction.Type.ChangeMultisigWallet || type == (int)Transaction.Type.MultisigAddTxSignature)
548  {
549  // pubkey must be one of the allowed signers on wallet
550  if(from_wallet.allowedSigners != null)
551  {
552  foreach(var allowed_signer in from_wallet.allowedSigners)
553  {
554  if(allowed_signer.SequenceEqual(p_address.address))
555  {
556  allowed = true;
557  }
558  }
559  }
560  }
561 
562  if (!allowed) return false;
563 
564  if(signature == null || pubkey == null)
565  {
566  Logging.warn("Signature or pubkey for received txid {0} was null, verification failed.", id);
567  return false;
568  }
569 
570  // Verify the signature
571  return CryptoManager.lib.verifySignature(checksum, pubkey, signature);
572  }
