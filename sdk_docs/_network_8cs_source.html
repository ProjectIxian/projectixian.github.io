184  public static IPv4Subnet FromCIDR(String cidr)
185  {
186  int delim = cidr.IndexOf('/');
187  if (delim == -1)
188  {
189  throw new ArgumentException(String.Format("Invalid CIDR notation: {0}", cidr));
190  }
191  string subnet = cidr.Substring(0, delim);
192  string mask = cidr.Substring(delim + 1);
193  IPAddress s;
194  if (IPAddress.TryParse(subnet, out s) == false)
195  {
196  throw new ArgumentException(String.Format("Invalid IP address: {0}", subnet));
197  }
198  if (s.AddressFamily != AddressFamily.InterNetwork)
199  {
200  throw new ArgumentException(String.Format("IPv4 address required: {0}", subnet));
201  }
202  int maskBits;
203  if (int.TryParse(mask, out maskBits) == false)
204  {
205  throw new ArgumentException(String.Format("Mask bits is not a number: {0}", mask));
206  }
207  if (maskBits < 0 || maskBits > 32)
208  {
209  throw new ArgumentException(String.Format("Invalid mask bits: {0}", mask));
210  }
211  uint mask_ip = 0xFFFFFFFF << (32 - maskBits);
212  byte[] maskBytes = new byte[]
213  {
214  (byte)((mask_ip & 0xFF000000) >> 24),
215  (byte)((mask_ip & 0x00FF0000) >> 16),
216  (byte)((mask_ip & 0x0000FF00) >> 8),
217  (byte)((mask_ip & 0x000000FF))
218  };
219  return new IPv4Subnet(s, new IPAddress(maskBytes));
220  }

310  public bool IsIPInSubnet(IPAddress addr)
311  {
312  if (addr.AddressFamily != AddressFamily.InterNetwork)
313  {
314  throw new ArgumentException(String.Format("IPv4 address required: {0}", addr.ToString()));
315  }
316  byte[] addressBytes = addr.GetAddressBytes();

318  byte[] maskBytes = mask.GetAddressBytes();

322  {
