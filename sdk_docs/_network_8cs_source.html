124  public static IPv4Subnet FromCIDR(String cidr)
125  {
126  int delim = cidr.IndexOf('/');
127  if(delim == -1)
128  {
129  throw new ArgumentException(String.Format("Invalid CIDR notation: {0}", cidr));
130  }
131  string subnet = cidr.Substring(0, delim);
132  string mask = cidr.Substring(delim + 1);
133  IPAddress s;
134  if(IPAddress.TryParse(subnet, out s) == false)
135  {
136  throw new ArgumentException(String.Format("Invalid IP address: {0}", subnet));
137  }
138  if(s.AddressFamily != AddressFamily.InterNetwork)
139  {
140  throw new ArgumentException(String.Format("IPv4 address required: {0}", subnet));
141  }
142  int maskBits;
143  if(int.TryParse(mask, out maskBits) == false)
144  {
145  throw new ArgumentException(String.Format("Mask bits is not a number: {0}", mask));
146  }
147  if (maskBits < 0 || maskBits > 32)
148  {
149  throw new ArgumentException(String.Format("Invalid mask bits: {0}", mask));
150  }
151  uint mask_ip = 0xFFFFFFFF << (32 - maskBits);
152  byte[] maskBytes = new byte[]
153  {
154  (byte)((mask_ip & 0xFF000000) >> 24),
155  (byte)((mask_ip & 0x00FF0000) >> 16),
156  (byte)((mask_ip & 0x0000FF00) >> 8),
157  (byte)((mask_ip & 0x000000FF))
158  };
159  return new IPv4Subnet(s, new IPAddress(maskBytes));
160  }

228  public bool IsIPInSubnet(IPAddress addr)
229  {
230  if(addr.AddressFamily != AddressFamily.InterNetwork)
231  {
232  throw new ArgumentException(String.Format("IPv4 address required: {0}", addr.ToString()));
233  }
234  byte[] addressBytes = addr.GetAddressBytes();
235  byte[] subnetBytes = subnet.GetAddressBytes();
236  byte[] maskBytes = mask.GetAddressBytes();
237  for(int i=0;i<addressBytes.Length;i++)
238  {
239  if((addressBytes[i] & maskBytes[i]) != (subnetBytes[i] & maskBytes[i]))
240  {
241  return false;
242  }
243  }
244  return true;
245  }
