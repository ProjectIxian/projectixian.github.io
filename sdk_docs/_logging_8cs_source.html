245  while (running)

285  public static void roll(bool forceRoll = false)
286  {
287  try
288  {
289  if (logFileStream != null && currentLogSize < maxLogSize)
290  {
291  return;
292  }
293 
294  if (File.Exists(logfilepath))
295  {
296  var length = new FileInfo(logfilepath).Length;
297  if (length > maxLogSize || (length > 0 && forceRoll))
298  {
299  if (logFileStream != null)
300  {
301  logFileStream.Flush();
302  logFileStream.Close();
303  logFileStream = null;
304  currentLogSize = 0;
305  }
306  string[] logFileList = Directory.GetFiles(folderpath, wildcard, SearchOption.TopDirectoryOnly);
307  if (logFileList.Length > 0)
308  {
309  // + 2 because of the . and digit [0-9]
310  var rolledLogFileList = logFileList.Where(fileName => Path.GetFileName(fileName).Length == (logfilename.Length + 2)).ToArray();
311  Array.Sort(rolledLogFileList, 0, rolledLogFileList.Length);
312  if (rolledLogFileList.Length >= maxLogCount)
313  {
314  File.Delete(rolledLogFileList[9]);
315  var list = rolledLogFileList.ToList();
316  list.RemoveAt(9);
317  rolledLogFileList = list.ToArray();
318  }
319 
320  // Move remaining rolled files
321  for (int i = rolledLogFileList.Length; i > 0; --i)
322  {
323  File.Move(rolledLogFileList[i - 1], logfilepathpart + "." + i + Path.GetExtension(logfilename));
324  }
325 
326  // Move original file
327  var targetPath = logfilepathpart + ".0" + Path.GetExtension(logfilename);
328  File.Move(logfilepath, targetPath);
329  }
330  }
331  }
332  }
333  catch (Exception e)
334  {
335  if (consoleOutput)
336  Console.WriteLine("Exception rolling log file: {0}", e.Message);
337  return;
338  }
339  try
340  {
341  if (logFileStream == null)
342  {
343  logFileStream = File.Open(logfilepath, FileMode.OpenOrCreate, FileAccess.Write, FileShare.Read);
344  }
345  }
346  catch (Exception e)
347  {
348  if (consoleOutput)
349  Console.WriteLine("Exception opening log file: {0}", e.Message);
350  }
