263  //spanish is at 2
264  languageCount[2]++;
265  }
266 
267  if (Wordlist.ChineseSimplified.WordExists(s, out index))
268  {
269  //chinese simplified is at 3
270  languageCount[3]++;
271  }
272 
273  if (Wordlist.ChineseTraditional.WordExists(s, out index) && !Wordlist.ChineseSimplified.WordExists(s, out index))

277  }
278  if (Wordlist.French.WordExists(s, out index))
279  {
280  languageCount[5]++;
281  }
282  }
283 
284  //no hits found for any language unknown
285  if (languageCount.Max() == 0)
286  {
287  return Language.Unknown;
288  }
289 
290  if (languageCount.IndexOf(languageCount.Max()) == 0)
291  {
292  return Language.English;
293  }
294  else if (languageCount.IndexOf(languageCount.Max()) == 1)
295  {
296  return Language.Japanese;
297  }
298  else if (languageCount.IndexOf(languageCount.Max()) == 2)

302  else if (languageCount.IndexOf(languageCount.Max()) == 3)

312  else if (languageCount.IndexOf(languageCount.Max()) == 4)
